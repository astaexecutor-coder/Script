-- CONFIG SAVE/LOAD SYSTEM
local FileName = "SlimeRNG_FinalConfig.json"

local function SaveConfig()
    local json = HttpService:JSONEncode(getgenv().Config)
    writefile(FileName, json)
end

local function LoadConfig()
    if isfile(FileName) then
        local content = readfile(FileName)
        local decoded = HttpService:JSONDecode(content)
        for k, v in pairs(decoded) do getgenv().Config[k] = v end
    end
end

-- REMOTE PATH HELPER
local function getRemotePath()
    return ReplicatedStorage:WaitForChild("Packages"):WaitForChild("_Index"):WaitForChild("leifstout_networker@0.3.1"):WaitForChild("networker"):WaitForChild("_remotes")
end

                if loot and root then
                    for _, drop in ipairs(loot:GetChildren()) do
                        for _, child in ipairs(drop:GetChildren()) do

-- COMBAT & SLIME TAB
-- ==========================================
Combat.newLabel("Attack & Radius")
Combat.newInput("Kill Radius", "Example: 300", function(v)
    getgenv().Config.KillRadius = tonumber(v) or 300
end)

Combat.newToggle("Slime Kill Aura", "Auto attack nearby enemies", false, function(v)
    getgenv().Config.KillAura = v
    task.spawn(function()
        local remote = getRemotePath():WaitForChild("CombatService"):WaitForChild("RemoteFunction")
        while getgenv().Config.KillAura do
            pcall(function()
                local root = Player.Character.HumanoidRootPart
                for _, enemy in ipairs(workspace:GetChildren()) do
                    if enemy:FindFirstChild("Humanoid") and (root.Position - enemy.HumanoidRootPart.Position).Magnitude <= getgenv().Config.KillRadius then
                        remote:InvokeServer("requestAttack", enemy)
                        if getgenv().Config.FastSlimeAttack then remote:InvokeServer("requestAttack", enemy) end
                    end
                end
            end)
            task.wait(getgenv().Config.FastSlimeAttack and 0.05 or 0.2)
        end
    end)
end)

Combat.newToggle("Fast Slime Attack", "No Attack Animation", false, function(v)

Upgrades.newToggle("Auto Teleport Best", "TP to highest zone", false, function(v)
    getgenv().Config.AutoTeleportBest = v
    task.spawn(function()
        local remote = getRemotePath():WaitForChild("ZonesService"):WaitForChild("RemoteFunction")
        while getgenv().Config.AutoTeleportBest do
            pcall(function()
                local best = 0
                for _, z in ipairs(workspace.Zones:GetChildren()) do
                    local n = tonumber(z.Name)
                    if n and z:FindFirstChild("Gate") and z.Gate:FindFirstChild("ClientGateBlocker_"..z.Name) and z.Gate["ClientGateBlocker_"..z.Name].CanCollide == false then
                        if n > best then best = n end
                    end
                end
                remote:InvokeServer("requestTeleportZone", best + 1)
            end)
            task.wait(10)
        end
    end)
end)

Upgrades.newToggle("Auto Upgrade Tree", "Dynamic Buy Upgrades", false, function(v)
    getgenv().Config.AutoUpgrade = v
    task.spawn(function()
        local remote = getRemotePath():WaitForChild("UpgradeService"):WaitForChild("RemoteFunction")
        while getgenv().Config.AutoUpgrade do
            pcall(function()
                local frame = Player.PlayerGui.Root.UpgradeScreen.UpgradeContent.Frame
                for _, t in ipairs(frame:GetChildren()) do
                    local id = t.Name:match("^(%S+)Tile")
                    if id then remote:InvokeServer("requestUnlock", id) end
                end
            end)
            task.wait(5)
        end
    end)
end)
